package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait FriendsFollowers { self: Http with Parameters =>
  def followersIds(userId: Either[UserId, ScreenName], cursor: Cursor = null, stringifyIds: StringifyIds = null)(implicit tokens: Tokens) = resource[Ids](get, "https://api.twitter.com/1/followers/ids.json", tokens, userId, cursor, stringifyIds)
  def existsFriendships(userIdA: Either[UserIdA, ScreenNameA], userIdB: Either[UserIdB, ScreenNameB])(implicit tokens: Option[Tokens]) = resource[Boolean](get, "https://api.twitter.com/1/friendships/exists.json", tokens, userIdA, userIdB)
  def incomingFriendships(cursor: Cursor = null, stringifyIds: StringifyIds = null)(implicit tokens: Tokens) = resource[Ids](get, "https://api.twitter.com/1/friendships/incoming.json", tokens, cursor, stringifyIds)
  def outgoingFriendships(cursor: Cursor = null, stringifyIds: StringifyIds = null)(implicit tokens: Tokens) = resource[Ids](get, "https://api.twitter.com/1/friendships/outgoing.json", tokens, cursor, stringifyIds)
  def showFriendships(sourceId: Either[SourceId, SourceScreenName], targetId: Either[TargetId, TargetScreenName])(implicit tokens: Option[Tokens]) = resource[Relationship](get, "https://api.twitter.com/1/friendships/show.json", tokens, sourceId, targetId)
  def createFriendships(userId: Either[UserId, ScreenName], follow: Follow = null)(implicit tokens: Tokens) = resource[User](post, "https://api.twitter.com/1/friendships/create.json", tokens, userId, follow)
  def destroyFriendships(userId: Either[UserId, ScreenName], includeEntities: IncludeEntities = null)(implicit tokens: Tokens) = resource[User](post, "https://api.twitter.com/1/friendships/destroy.json", tokens, userId, includeEntities)
  def lookupFriendships(userId: Either[UserId, ScreenName])(implicit tokens: Tokens) = resource[Friendship](get, "https://api.twitter.com/1/friendships/lookup.json", tokens, userId)
  def updateFriendships(userId: Either[UserId, ScreenName], device: Device = null, retweets: Retweets = null)(implicit tokens: Tokens) = resource[Relationship](post, "https://api.twitter.com/1/friendships/update.json", tokens, userId, device, retweets)
  def noRetweetIds(stringifyIds: StringifyIds = null)(implicit tokens: Tokens) = resource[List[Long]](get, "https://api.twitter.com/1/friendships/no_retweet_ids.json", tokens, stringifyIds)
}

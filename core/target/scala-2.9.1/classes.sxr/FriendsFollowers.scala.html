package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait FriendsFollowers { self: Parameters =>
  def followersIds(userId: Either[UserId, ScreenName], cursor: Cursor = null, stringifyIds: StringifyIds = null)(implicit tokens: Some[Tokens]) = resource[Ids](get, "followers/ids", tokens, userId, cursor, stringifyIds)
  def existsFriendships(userIdA: Either[UserIdA, ScreenNameA], userIdB: Either[UserIdB, ScreenNameB])(implicit tokens: Option[Tokens]) = resource[Boolean](get, "friendships/exists", tokens, userIdA, userIdB)
  def incomingFriendships(stringifyIds: StringifyIds = null)(implicit tokens: Some[Tokens]) = resource[Ids](get, "cursor", tokens, stringifyIds)
  def outgoingFriendships(stringifyIds: StringifyIds = null)(implicit tokens: Some[Tokens]) = resource[Ids](get, "cursor", tokens, stringifyIds)
  def showFriendships(sourceId: Either[SourceId, SourceScreenName], targetId: Either[TargetId, TargetScreenName])(implicit tokens: Option[Tokens]) = resource[Relationship](get, "friendships/show", tokens, sourceId, targetId)
  def createFriendships(userId: Either[UserId, ScreenName], follow: Follow = null)(implicit tokens: Some[Tokens]) = resource[User](post, "friendships/create", tokens, userId, follow)
  def destroyFriendships(userId: Either[UserId, ScreenName], includeEntities: IncludeEntities = null)(implicit tokens: Some[Tokens]) = resource[User](post, "friendships/destroy", tokens, userId, includeEntities)
  def lookupFriendships(userId: Either[UserId, ScreenName])(implicit tokens: Some[Tokens]) = resource[Friendship](get, "friendships/lookup", tokens, userId)
  def updateFriendships(userId: Either[UserId, ScreenName], device: Device = null, retweets: Retweets = null)(implicit tokens: Some[Tokens]) = resource[Relationship](post, "friendships/update", tokens, userId, device, retweets)
  def noRetweetIds(stringifyIds: StringifyIds = null)(implicit tokens: Some[Tokens]) = resource[List[Long]](get, "friendships/no_retweet_ids", tokens, stringifyIds)
}

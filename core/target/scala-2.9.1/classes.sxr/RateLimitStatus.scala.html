package twitter4z.objects

import net.liftweb.json.scalaz.JsonScalaz._

case class RateLimitStatus(
  remainingHits: Int,
  resetTimeInSeconds: Long,
  hourlyLimit: Int,
  resetTime: Int
)

object RateLimitStatus {
  implicit def RateLimitStatusJSONR: JSONR[RateLimitStatus] = (RateLimitStatus.apply _).applyJSON(field("remaining_hits"), field("reset_time_in_seconds"), field("hourly_limit"), field("reset_time"))
}

package twitter4z.objects

import net.liftweb.json.scalaz.JsonScalaz._

case class TimeZone(
  name: String,
  tzinfoName: String,
  utcOffset: Long
)

object TimeZone {
  implicit def TimeZoneJSONR: JSONR[TimeZone] = (TimeZone.apply _).applyJSON(field("name"), field("tzinfo_name"), field("utc_offset"))
}

package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait Accounts { self: Http with Parameters =>
  def rateLimitStatus()(implicit tokens: OptionTokens) = resource[RateLimitStatus](get, "https://api.twitter.com/1/account/rate_limit_status.json", tokens)
  def verifyCredentials(includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: Tokens) = resource[User](get, "https://api.twitter.com/1/account/verify_credentials.json", tokens, includeEntities, skipStatus)
  def endSession()(implicit tokens: Tokens) = resource[EndSessionRequest](post, "https://api.twitter.com/1/account/end_session.json", tokens)
  def updateProfile(name: Name = null, url: Url = null, location: Location = null, description: Description = null, includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: Tokens) = resource[User](post, "https://api.twitter.com/1/account/update_profile.json", tokens, name, url, location, description, includeEntities, skipStatus)
  def updateProfileBackgroundImage(image: Image = null, tile: Tile = null, includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null, use: Use = null)(implicit tokens: Tokens) = resource[User](post, "https://api.twitter.com/1/account/update_profile_background_image.json", tokens, image, tile, includeEntities, skipStatus, use)
  def updateProfileColors(profileBackgroundColor: ProfileBackgroundColor = null, profileLinkColor: ProfileLinkColor = null, profileSidebarBorderColor: ProfileSidebarBorderColor = null, profileSidebarFillColor: ProfileSidebarFillColor = null, profileTextColor: ProfileTextColor = null, includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: Tokens) = resource[User](post, "https://api.twitter.com/1/account/update_profile_colors.json", tokens, profileBackgroundColor, profileLinkColor, profileSidebarBorderColor, profileSidebarFillColor, profileTextColor, includeEntities, skipStatus)
  def updateProfileImage(image: Image, includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: Tokens) = resource[User](post, "https://api.twitter.com/1/account/update_profile_image.json", tokens, image, includeEntities, skipStatus)
  def totals()(implicit tokens: Tokens) = resource[Totals](get, "https://api.twitter.com/1/account/totals.json", tokens)
  def accountSettings()(implicit tokens: Tokens) = resource[AccountSettings](get, "https://api.twitter.com/1/account/settings.json", tokens)
  def updateAccountSettings(trendLocationWoeid: TrendLocationWoeid = null, sleepTimeEnabled: SleepTimeEnabled = null, startSleepTime: StartSleepTime = null, endSleepTime: EndSleepTime = null, timeZone: TimeZone = null, lang: Lang = null)(implicit tokens: Tokens) = resource[AccountSettings](post, "https://api.twitter.com/1/account/settings.json", tokens, trendLocationWoeid, sleepTimeEnabled, startSleepTime, endSleepTime, timeZone, lang)
}

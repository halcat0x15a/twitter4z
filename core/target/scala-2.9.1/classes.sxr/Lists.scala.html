package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait Lists { self: Http with Parameters =>
  def allLists(userId: Either[UserId, ScreenName] = null)(implicit tokens: Option[Tokens]) = resource[List[UserList]](get, "https://api.twitter.com/1/lists/all.json", tokens, userId)
  def listStatuses(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])], sinceId: SinceId = null, maxId: MaxId = null, perPage: PerPage = null, page: Page = null, includeEntities: IncludeEntities = null, includeRts: IncludeRts = null)(implicit tokens: Option[Tokens]) = resource[List[StatusObject]](get, "https://api.twitter.com/1/lists/statuses.json", tokens, listId, sinceId, maxId, perPage, page, includeEntities, includeRts)
  def destroyListMember(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])], userId: Either[UserId, ScreenName])(implicit tokens: Tokens) = resource[UserList](post, "https://api.twitter.com/1/lists/members/destroy.json", tokens, listId, userId)
  def memberships(userId: Either[UserId, ScreenName] = null, cursor: Cursor = null, filterToOwnedLists: FilterToOwnedLists = null)(implicit tokens: Option[Tokens]) = resource[UserLists](get, "https://api.twitter.com/1/lists/memberships.json", tokens, userId, cursor, filterToOwnedLists)
  def subscribers(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])], cursor: Cursor = null, includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: Option[Tokens]) = resource[UserLists](get, "https://api.twitter.com/1/lists/subscribers.json", tokens, listId, cursor, includeEntities, skipStatus)
  def createSubscribers(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])])(implicit tokens: Tokens) = resource[UserList](post, "https://api.twitter.com/1/lists/subscribers/create.json", tokens, listId)
  def showSubscribers(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])], userId: Either[UserId, ScreenName], includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: Tokens) = resource[User](get, "https://api.twitter.com/1/lists/subscribers/show.json", tokens, listId, userId, includeEntities, skipStatus)
  def destroySubscribers(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])])(implicit tokens: Tokens) = resource[UserList](post, "https://api.twitter.com/1/lists/subscribers/destroy.json", tokens, listId)
  def createAllMembers(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])], userId: Either[UserId, ScreenName])(implicit tokens: Tokens) = resource[UserList](post, "https://api.twitter.com/1/lists/members/create_all.json", tokens, listId, userId)
  def showMembers(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])], userId: Either[UserId, ScreenName], includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: Tokens) = resource[User](get, "https://api.twitter.com/1/lists/members/show.json", tokens, listId, userId, includeEntities, skipStatus)
  def members(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])], cursor: Cursor = null, includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: Option[Tokens]) = resource[UserLists](get, "https://api.twitter.com/1/lists/members.json", tokens, listId, cursor, includeEntities, skipStatus)
  def createMember(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])], userId: Either[UserId, ScreenName])(implicit tokens: Tokens) = resource[UserList](post, "https://api.twitter.com/1/lists/members/create.json", tokens, listId, userId)
  def destroyList(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])])(implicit tokens: Tokens) = resource[UserList](post, "https://api.twitter.com/1/lists/destroy.json", tokens, listId)
  def updateList(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])], name: Name = null, mode: Mode = null, description: Description = null)(implicit tokens: Tokens) = resource[UserList](post, "https://api.twitter.com/1/lists/update.json", tokens, listId, name, mode, description)
  def createList(name: Name, mode: Mode = null, description: Description = null)(implicit tokens: Tokens) = resource[UserList](post, "https://api.twitter.com/1/lists/create.json", tokens, name, mode, description)
  def lists(userId: Either[UserId, ScreenName], cursor: Cursor = null)(implicit tokens: Tokens) = resource[UserLists](get, "https://api.twitter.com/1/lists.json", tokens, userId, cursor)
  def showLists(listId: Either[ListId, (Slug, Either[OwnerId, OwnerScreenName])])(implicit tokens: Option[Tokens]) = resource[UserLists](get, "https://api.twitter.com/1/lists/show.json", tokens, listId)
}

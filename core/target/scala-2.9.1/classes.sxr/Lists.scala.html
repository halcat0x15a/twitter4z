package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait Lists { self: Parameters =>
  def allLists(userId: Either[UserId, ScreenName] = null)(implicit tokens: Option[Tokens]) = resource[List[UserList]](get, "lists/all", tokens, userId)
  def listStatuses(listId: Either[ListId, Slug], ownerScreenName: OwnerScreenName = null, ownerId: OwnerId = null, sinceId: SinceId = null, maxId: MaxId = null, perPage: PerPage = null, page: Page = null, includeEntities: IncludeEntities = null, includeRts: IncludeRts = null)(implicit tokens: Option[Tokens]) = resource[List[StatusObject]](get, "lists/statuses", tokens, listId, ownerScreenName, ownerId, sinceId, maxId, perPage, page, includeEntities, includeRts)
  def destroyListMember(listId: Either[ListId, Slug], userId: Either[UserId, ScreenName], ownerScreenName: OwnerScreenName = null, ownerId: OwnerId = null)(implicit tokens: Some[Tokens]) = resource[UserList](post, "lists/members/destroy", tokens, listId, userId, ownerScreenName, ownerId)
}

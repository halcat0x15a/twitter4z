package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait Tweets { self: Http with Parameters =>
  def retweetedBy(id: Id, count: Count = null, page: Page = null)(implicit tokens: Option[Tokens]) = resource[Users](get, "https://api.twitter.com/1/statuses/" + id + "/retweeted_by.json", tokens, count, page)
  def retweetedByIds(id: Id, count: Count = null, page: Page = null, stringifyIds: StringifyIds = null)(implicit tokens: Tokens) = resource[List[Long]](get, "https://api.twitter.com/1/statuses/" + id + "/retweeted_by/ids.json", tokens, count, page, stringifyIds)
  def retweets(id: Id, count: Count = null, trimUser: TrimUser = null, includeEntities: IncludeEntities = null)(implicit tokens: Tokens) = resource[Statuses](get, "https://api.twitter.com/1/statuses/retweets/" + id + ".json", tokens, count, trimUser, includeEntities)
  def showStatus(id: Id, trimUser: TrimUser = null, includeEntities: IncludeEntities = null) = resource[StatusObject](get, "https://api.twitter.com/1/statuses/show/" + id + ".json", None, trimUser, includeEntities)
  def destroyStatus(id: Id, includeEntities: IncludeEntities = null, trimUser: TrimUser = null)(implicit tokens: Tokens) = resource[StatusObject](post, "https://api.twitter.com/1/statuses/destroy/" + id + ".json", tokens, includeEntities, trimUser)
  def retweet(id: Id, includeEntities: IncludeEntities = null, trimUser: TrimUser = null)(implicit tokens: Tokens) = resource[StatusObject](post, "https://api.twitter.com/1/statuses/retweet/" + id + ".json", tokens, includeEntities, trimUser)
  def updateStatus(status: Status, inReplyToStatusId: InReplyToStatusId = null, latlong: (Latitude, Longitude) = null, placeId: PlaceId = null, displayCoordinates: DisplayCoordinates = null, trimUser: TrimUser = null, includeEntities: IncludeEntities = null, wrapLinks: WrapLinks = null)(implicit tokens: Tokens) = resource[StatusObject](post, "https://api.twitter.com/1/statuses/update.json", tokens, status, inReplyToStatusId, latlong, placeId, displayCoordinates, trimUser, includeEntities, wrapLinks)
}

package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait Tweets { self: Parameters =>
  def retweetedBy(id: Id, count: Count = null, page: Page = null)(implicit tokens: Option[Tokens]) = resource[Users](get, "statuses/" + id + "/retweeted_by", tokens, count, page)
  def retweetedByIds(id: Id, count: Count = null, page: Page = null, stringifyIds: StringifyIds = null)(implicit tokens: Some[Tokens]) = resource[List[Long]](get, "statuses/" + id + "/retweeted_by/ids", tokens, count, page, stringifyIds)
  def retweets(id: Id, count: Count = null, trimUser: TrimUser = null, includeEntities: IncludeEntities = null)(implicit tokens: Some[Tokens]) = resource[Statuses](get, "statuses/retweets/" + id + "", tokens, count, trimUser, includeEntities)
  def showStatus(id: Id, trimUser: TrimUser = null, includeEntities: IncludeEntities = null) = resource[StatusObject](get, "statuses/show/" + id + "", None, trimUser, includeEntities)
  def destroyStatus(id: Id, includeEntities: IncludeEntities = null, trimUser: TrimUser = null)(implicit tokens: Some[Tokens]) = resource[StatusObject](post, "statuses/destroy/" + id + "", tokens, includeEntities, trimUser)
  def retweet(id: Id, includeEntities: IncludeEntities = null, trimUser: TrimUser = null)(implicit tokens: Some[Tokens]) = resource[StatusObject](post, "statuses/retweet/" + id + "", tokens, includeEntities, trimUser)
  def update(status: Status, inReplyToStatusId: InReplyToStatusId = null, lat: Latitude = null, long: Longitude = null, placeId: PlaceId = null, displayCoordinates: DisplayCoordinates = null, trimUser: TrimUser = null, includeEntities: IncludeEntities = null, wrapLinks: WrapLinks = null)(implicit tokens: Some[Tokens]) = resource[StatusObject](post, "statuses/update", tokens, status, inReplyToStatusId, lat, long, placeId, displayCoordinates, trimUser, includeEntities, wrapLinks)
}

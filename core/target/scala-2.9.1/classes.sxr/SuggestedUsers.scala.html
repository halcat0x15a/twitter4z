package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait SuggestedUsers { self: Http with Parameters =>
  def suggestions(lang: Lang = null) = resource[Category](get, "https://api.twitter.com/1/users/suggestions.json", NoneTokens, lang)
  def suggestionUsers(slug: Slug, lang: Lang = null) = resource[Category](get, "https://api.twitter.com/1/users/suggestions/" + slug + ".json", NoneTokens, lang)
  def suggestionMembers(slug: Slug) = resource[List[User]](get, "https://api.twitter.com/1/users/suggestions/" + slug + "/members.json", NoneTokens)
}

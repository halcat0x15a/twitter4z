package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait SuggestedUsers { self: Parameters =>
  def suggestions(lang: Lang = null) = resource[Category](get, "users/suggestions", None, lang)
  def suggestionUsers(slug: Slug, lang: Lang = null) = resource[Category](get, "users/suggestions/" + slug + "", None, slug, lang)
  def suggestionMembers(slug: Slug) = resource[List[User]](get, "users/suggestions/" + slug + "/members", None, slug)
}

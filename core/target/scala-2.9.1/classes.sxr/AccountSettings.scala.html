package twitter4z.objects

import net.liftweb.json.scalaz.JsonScalaz._

case class AccountSettings(
  sleepTime: SleepTime,
  trendLocation: Location,
  language: String,
  alwaysUseHttps: Boolean,
  discoverableByEmail: Boolean,
  timeZone: TimeZone,
  geoEnabled: Boolean
)

object AccountSettings {
  implicit def AccountSettingsJSONR: JSONR[AccountSettings] = (AccountSettings.apply _).applyJSON(field("sleep_time"), field("trend_location"), field("language"), field("always_use_https"), field("discoverable_by_email"), field("time_zone"), field("geo_enabled"))
}

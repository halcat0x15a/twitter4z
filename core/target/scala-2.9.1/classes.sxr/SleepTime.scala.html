package twitter4z.objects

import net.liftweb.json.scalaz.JsonScalaz._

case class SleepTime(
  startTime: String,
  endTime: String,
  enabled: Boolean
)

object SleepTime {
  implicit def SleepTimeJSONR: JSONR[SleepTime] = (SleepTime.apply _).applyJSON(field("start_time"), field("end_time"), field("enabled"))
}

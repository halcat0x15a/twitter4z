package twitter4z.objects

import net.liftweb.json.scalaz.JsonScalaz._

case class Totals(
  friends: Int,
  updates: Int,
  followers: Int,
  favorites: Int
)

object Totals {
  implicit def TotalsJSONR: JSONR[Totals] = (Totals.apply _).applyJSON(field("friends"), field("updates"), field("followers"), field("favorites"))
}

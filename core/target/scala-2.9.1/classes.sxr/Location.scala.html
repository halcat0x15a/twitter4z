package twitter4z.objects

import net.liftweb.json.scalaz.JsonScalaz._

case class Location(
  name: String,
  woeid: Int,
  placeType: PlaceType,
  country: String,
  url: String,
  countryCode: String
)

object Location {
  implicit def LocationJSONR: JSONR[Location] = (Location.apply _).applyJSON(field("name"), field("woeid"), field("placeType"), field("country"), field("url"), field("countryCode"))
}

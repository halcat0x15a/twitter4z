package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait Favorites { self: Parameters =>
  def favorites(userId: Either[UserId, ScreenName] = null, sinceId: SinceId = null, page: Page = null, includeEntities: IncludeEntities = null)(implicit tokens: Some[Tokens]) = resource[List[StatusObject]](get, "favorites", tokens, userId, sinceId, page, includeEntities)
  def createFavorite(id: Id, includeEntities: IncludeEntities = null)(implicit tokens: Some[Tokens]) = resource[StatusObject](post, "favorites/create/" + id + "", tokens, includeEntities)
  def destroyFavorite(id: Id)(implicit tokens: Some[Tokens]) = resource[StatusObject](post, "favorites/destroy/" + id + "", tokens)
}

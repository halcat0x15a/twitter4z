package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait Favorites { self: Http with Parameters =>
  def favorites(userId: Either[UserId, ScreenName], sinceId: SinceId = null, page: Page = null, includeEntities: IncludeEntities = null)(implicit tokens: Tokens) = resource[List[StatusObject]](get, "https://api.twitter.com/1/favorites.json", tokens, userId, sinceId, page, includeEntities)
  def createFavorite(id: Id, includeEntities: IncludeEntities = null)(implicit tokens: Tokens) = resource[StatusObject](post, "https://api.twitter.com/1/favorites/create/" + id + ".json", tokens, includeEntities)
  def destroyFavorite(id: Id)(implicit tokens: Tokens) = resource[StatusObject](post, "https://api.twitter.com/1/favorites/destroy/" + id + ".json", tokens)
}

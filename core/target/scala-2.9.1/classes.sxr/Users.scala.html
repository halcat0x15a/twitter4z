package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait Users { self: Http with Parameters =>
  def lookupUsers(userId: Either[UserId, ScreenName], includeEntities: IncludeEntities = null)(implicit tokens: OptionTokens) = resource[Users](get, "https://api.twitter.com/1/users/lookup.json", tokens, userId, includeEntities)
  def profileImage(screenName: ScreenName, size: Size = null) = resource[URL](get, "https://api.twitter.com/1/users/profile_image/" + screenName + ".json", NoneTokens, screenName, size)
  def searchUsers(q: Query, page: Page = null, perPage: PerPage = null, includeEntities: IncludeEntities = null)(implicit tokens: Tokens) = resource[Users](get, "https://api.twitter.com/1/users/search.json", tokens, q, page, perPage, includeEntities)
  def showUser(userId: Either[UserId, ScreenName], includeEntities: IncludeEntities = null)(implicit tokens: OptionTokens) = resource[User](get, "https://api.twitter.com/1/users/show.json", tokens, userId, includeEntities)
  def contributees(userId: Either[UserId, ScreenName], includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: OptionTokens) = resource[Users](get, "https://api.twitter.com/1/users/contributees.json", tokens, userId, includeEntities, skipStatus)
  def contributors(userId: Either[UserId, ScreenName], includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: OptionTokens) = resource[Users](get, "https://api.twitter.com/1/users/contributors.json", tokens, userId, includeEntities, skipStatus)
}

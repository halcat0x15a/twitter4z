package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait Users { self: Parameters =>
  def lookupUsers(userId: Either[UserId, ScreenName], includeEntities: IncludeEntities = null)(implicit tokens: Option[Tokens]) = resource[Users](get, "users/lookup", tokens, userId, includeEntities)
  def profileImage(screenName: ScreenName, size: Size = null) = resource[URL](get, "users/profile_image/" + screenName + "", None, size)
  def searchUsers(q: Query, page: Page = null, perPage: PerPage = null, includeEntities: IncludeEntities = null)(implicit tokens: Some[Tokens]) = resource[Users](get, "users/search", tokens, q, page, perPage, includeEntities)
  def showUser(userId: Either[UserId, ScreenName], includeEntities: IncludeEntities = null)(implicit tokens: Option[Tokens]) = resource[User](get, "users/show", tokens, userId, includeEntities)
  def contributees(userId: Either[UserId, ScreenName], includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: Option[Tokens]) = resource[Users](get, "users/contributees", tokens, userId, includeEntities, skipStatus)
  def contributors(userId: Either[UserId, ScreenName], includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: Option[Tokens]) = resource[Users](get, "users/contributors", tokens, userId, includeEntities, skipStatus)
}

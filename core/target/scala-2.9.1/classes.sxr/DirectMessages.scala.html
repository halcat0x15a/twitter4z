package twitter4z.api

import twitter4z.objects._
import twitter4z.http._
import twitter4z.json._
import twitter4z.auth._

trait DirectMessages { self: Http with Parameters =>
  def directMessages(sinceId: SinceId = null, maxId: MaxId = null, count: Count = null, page: Page = null, includeEntities: IncludeEntities = null, skipStatus: SkipStatus = null)(implicit tokens: Tokens) = resource[DirectMessages](get, "https://api.twitter.com/1/direct_messages.json", tokens, sinceId, maxId, count, page, includeEntities, skipStatus)
  def sentDirectMessages(sinceId: SinceId = null, maxId: MaxId = null, count: Count = null, page: Page = null, includeEntities: IncludeEntities = null)(implicit tokens: Tokens) = resource[DirectMessages](get, "https://api.twitter.com/1/direct_messages/sent.json", tokens, sinceId, maxId, count, page, includeEntities)
  def destroyDirectMessage(id: Id, includeEntities: IncludeEntities = null)(implicit tokens: Tokens) = resource[DirectMessage](post, "https://api.twitter.com/1/direct_messages/destroy/" + id + ".json", tokens, includeEntities)
  def newDirectMessage(userId: Either[UserId, ScreenName], text: Text, wrapLinks: WrapLinks = null)(implicit tokens: Tokens) = resource[DirectMessage](post, "https://api.twitter.com/1/direct_messages/new.json", tokens, userId, text, wrapLinks)
  def showDirectMessage(id: Id)(implicit tokens: Tokens) = resource[DirectMessage](get, "https://api.twitter.com/1/direct_messages/show/" + id + ".json", tokens)
}
